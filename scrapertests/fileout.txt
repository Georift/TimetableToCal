function clsResizeDiv(mainContain, bodyPnl, isHighlight){
       if (bodyPnl.size() == 0) {return false;} 
       if (!isHighlight && mainContain.data('OrgHeight') != null) { 
           mainContain.height(mainContain.data('OrgHeight')); 
           if (mainContain.css('z-index') != '999') {mainContain.css('z-index', '');} 
           mainContain.removeData('OrgHeight')
           return true;
       }
       if (!isHighlight) {return false;}
       if (mainContain.data('OrgHeight') == null) { 
           mainContain.data('OrgHeight', mainContain.height());} 
       if (mainContain.css('z-index') != '999') {mainContain.css('z-index', '995');} 
       while (bodyPnl[0].scrollHeight > bodyPnl.height()) 
       {
           mainContain.height(mainContain.height() + 10); 
       }
    return false;
   }
